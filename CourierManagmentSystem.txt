

create table shipment ( 
sh_id int primary key auto_increment,
weight numeric not null , 
category varchar(50) not null, 
issue_date date not null ,
delivery_date date not null ,
status varchar(25) not null 
)auto_increment=20000;

create table shipper (
s_id int primary key auto_increment,
name varchar(30) not null,
address text,
contact varchar(11)
)auto_increment=21000;


create table courier(
c_id int primary key auto_increment,
name varchar(30) , 
status  text , 
PASSWORD varchar(15) not null , 
address text, 
contact varchar(11) )auto_increment=22000;

create table manager(
	m_id int primary key,
    name varchar(30) NOT null,
    password varchar(15) not null , 
    contact varchar(11) 
);

create table customer (
name varchar(30),
phone varchar(11),
address text ,
s_id int, 
FOREIGN key (s_id) REFERENCES shipper (s_id) on delete cascade ,
PRIMARY key (name, s_id)
);

create TABLE payment (
    p_id int primary key auto_increment,
    amount decimal(10,3),
    tax decimal(10,2) ,
    total decimal(10,3) ,
    issue_date timestamp 
)auto_increment=2023080001;

CREATE TABLE pay (
s_id int , FOREIGN key (s_id) REFERENCES shipper(s_id) on delete cascade,
    sh_id int , FOREIGN key (sh_id) REFERENCES shipment (sh_id) on delete cascade,
    p_id int , FOREIGN key (p_id) REFERENCES payment (p_id) on delete cascade,
    primary key(s_id,sh_id,p_id)
);
CREATE TABLE assign (
m_id int , FOREIGN KEY (m_id) REFERENCES manager (m_id) on delete cascade,
sh_id int , FOREIGN key (sh_id) REFERENCES shipment (sh_id) on delete cascade,
c_id int , FOREIGN key (c_id) REFERENCES courier (c_id) on delete cascade,
primary key(m_id,sh_id,c_id)
);
